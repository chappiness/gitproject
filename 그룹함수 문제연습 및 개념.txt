-- 1.그룹함수
SELECT AVG(SAL) FROM EMP;
SELECT ROUND(AVG(SAL),2) FROM EMP;
SELECT  COUNT(SAL) FROM EMP; -- EMP 테이블의 행수
SELECT AVG(SAL),SUM(SAL)/COUNT(SAL) FROM EMP;
SELECT COMM FROM EMP; --14개인데 NULL 포함 안함 (300+500+1400+0)/3
SELECT AVG(COMM) FROM EMP;--모든 그룹함수는 NULL 값을 제왜시킨다!
SELECT AVG(COMM)"평균",SUM(COMM)"합",COUNT(COMM)"갯수",SUM(COMM)/COUNT(COMM) "평균" FROM EMP;

-- SAL의 평균, 합, 최소값, 최대값, 분산, 표준편차, 갯수
SELECT ROUND(AVG(SAL),2),SUM(SAL),MIN(SAL),MAX(SAL),ROUND(VARIANCE(SAL)),ROUND(STDDEV(SAL)),
                                                    COUNT(*) FROM EMP;
SELECT SQRT(VARIANCE(SAL)),STDDEV(SAL) FROM EMP; --표준편차와 분산의 관계

-- 그룹함수 MAX, MIN, COUNT는 숫자형, 문자형, 날짜형 모두 사용가능
-- 가장 최근에 입사한 사원의 입사일과 입사한지 가장 최초입사한 사원의 입사일
SELECT HIREDATE FROM EMP;
SELECT MAX(HIREDATE) FROM EMP; --가장 최근의 입사일
SELECT MIN(HIREDATE) FROM EMP; --가장 오래된 직원의 입사일 
-- 83/01/12 최근입사 : X,XXX일 근무    80/12/17 최초입사 : XX,XXX일 근무
SELECT TO_CHAR(TRUNC(SYSDATE-MIN(HIREDATE)),'99,999') FROM EMP; --가장 오래된 직원 근무일수
SELECT TO_CHAR(TRUNC(SYSDATE-MAX(HIREDATE)),'99,999') FROM EMP; --가장 신참인 직원 근무일수
SELECT MAX(HIREDATE)||'최근입사:'||TO_CHAR(TRUNC(SYSDATE-MAX(HIREDATE)),'99,999')||'일 근무' FROM EMP;
SELECT MIN(HIREDATE)||'최근입사:'||TO_CHAR(TRUNC(SYSDATE-MAX(HIREDATE)),'99,999')||'일 근무' FROM EMP;

-- 10번 부서 소속의 사원 급여 평균

SELECT AVG(SAL) FROM EMP WHERE DEPTNO=30;

-- ★ 2. GROUP BY 절
-- 부서번호별 최대급여
SELECT DEPTNO,MAX(SAL) FROM EMP GROUP BY DEPTNO;

-- 부서번호별 사원수, 최대급여, 최소급여, 평균급여(소수점2자리)
SELECT DEPTNO,COUNT(*),MAX(SAL),MIN(SAL),ROUND(AVG(SAL),2) FROM EMP GROUP BY DEPTNO ORDER BY DEPTNO;

-- SAL이 5000미만인 사원에 대해서만 부서번호별 사원수, 최대급여, 최소급여, 평균급여
SELECT  DEPTNO,COUNT(*),MAX(SAL),MIN(SAL),AVG(SAL) FROM EMP WHERE SAL<5000 GROUP BY DEPTNO ORDER BY DEPTNO;
--COMM=NULL인 부서별 사원의 수
SELECT DEPTNO,COUNT(*) FROM EMP GROUP BY DEPTNO;
SELECT DEPTNO,COUNT(*) FROM EMP WHERE COMM IS NULL GROUP BY DEPTNO;
SELECT DEPTNO,COUNT(*) FROM EMP WHERE NOT COMM IS NULL GROUP BY DEPTNO;

-- 부서명별 사원수, 최대급여, 최소급여, 평균급여--JOIN 이용
SELECT DNAME,COUNT(*),MAX(SAL),MIN(SAL),ROUND(AVG(SAL),2) FROM EMP E,DEPT D 
                                                WHERE E.DEPTNO=D.DEPTNO
                                                GROUP BY DNAME; 
--3. HAVING 절 ; 그룹함수 결과의 조건
-- 부서번호별 평균 급여(평균급여가 2000이상인 부서만 출력)
--SELECT DEPTNO,AVG(SAL) FROM EMP WHERE AVG(SAL)>20000 (X)
SELECT DEPTNO,ROUND(AVG(SAL),2) FROM EMP GROUP BY DEPTNO HAVING AVG(SAL)>2000;

--4. 결과 집합 내에 집계값 생성
SELECT DEPTNO,SUM(SAL) FROM EMP GROUP BY ROLLUP(DEPTNO);-- 부서별 합계까지 출력
SELECT DEPTNO,JOB,SUM(SAL) FROM EMP GROUP BY ROLLUP(DEPTNO,JOB); --부서별,직업별 합계까지 출력

--7. 사원수가 5명이상 넘는 부서번호와 사원수를 출력

-- 8. 사원수가 3명이하인 부서명과 사원수를 출력
SELECT DNAME,COUNT(*) FROM EMP E,DEPT D WHERE E.DEPTNO=D.DEPTNO GROUP BY DNAME HAVING COUNT(*)<=3;

--9. 업무별 급여의 평균이 1000이상인 업무에 대해서 업무명, 평균 급여, 급여의 합을 출력
SELECT DNAME,AVG(SAL),SUM(SAL) FROM EMP E,DEPT D WHERE E.DEPTNO=D.DEPTNO 
                                            GROUP BY DNAME HAVING AVG(SAL)>1000;
                                            
--10. 급여 합이 5000을 초과하는 각 업무에 대해서 업무와 급여합을 출력(급여 합계순 내림차순 정렬)
SELECT JOB,SUM(SAL) FROM EMP GROUP BY JOB HAVING SUM(SAL)>5000 ORDER BY SUM(SAL) DESC; 

--11. 부서별 급여평균, 부서별 급여합계, 부서별 최소급여를 출력
SELECT DEPTNO,AVG(SAL),SUM(SAL),MIN(SAL) FROM EMP GROUP BY DEPTNO;

--12D. 위의 11번을 수정하여, 부서별 급여평균 최대치, 부서별 급여합의 최대치, 부서별 최소급여의 최대치를 출력
SELECT MAX(AVG(SAL)),MAX(SUM(SAL)),MAX(MIN(SAL)) FROM EMP GROUP BY DEPTNO;
